# Hill Cipher

def char_to_num(c):
    return ord(c) - 65

def num_to_char(n):
    return chr(n + 65)

def mod_inverse(det):
    det = det % 26
    for i in range(26):
        if (det * i) % 26 == 1:
            return i
    return -1

def encrypt(text, key):
    cipher = ""
    for i in range(0, len(text), 2):
        p1 = char_to_num(text[i])
        p2 = char_to_num(text[i + 1])
        c1 = (key[0][0]*p1 + key[0][1]*p2) % 26
        c2 = (key[1][0]*p1 + key[1][1]*p2) % 26
        cipher += num_to_char(c1) + num_to_char(c2)
    return cipher

def decrypt(cipher, key):
    det = (key[0][0]*key[1][1] - key[0][1]*key[1][0]) % 26
    inv_det = mod_inverse(det)
    inv_key = [
        [( key[1][1]*inv_det) % 26, (-key[0][1]*inv_det) % 26],
        [(-key[1][0]*inv_det) % 26, ( key[0][0]*inv_det) % 26]
    ]
    plain = ""
    for i in range(0, len(cipher), 2):
        c1 = char_to_num(cipher[i])
        c2 = char_to_num(cipher[i + 1])
        p1 = (inv_key[0][0]*c1 + inv_key[0][1]*c2) % 26
        p2 = (inv_key[1][0]*c1 + inv_key[1][1]*c2) % 26
        plain += num_to_char(p1) + num_to_char(p2)
    return plain

# Playfair Cipher

def generate_matrix(key):
    matrix = []
    used   = []
    for ch in key:
        if ch == 'J': ch = 'I'
        if ch not in used and 'A' <= ch <= 'Z':
            used.append(ch)
    for ch in "ABCDEFGHIKLMNOPQRSTUVWXYZ":
        if ch not in used: used.append(ch)
    k = 0
    for i in range(5):
        row = []
        for j in range(5):
            row.append(used[k]); k += 1
        matrix.append(row)
    return matrix

def display_matrix(matrix):
    print("\n5x5 Key Matrix:")
    for row in matrix:
        for ch in row:
            print(ch, end=" ")
        print()

def find_pos(matrix, ch):
    if ch == 'J': ch = 'I'
    for i in range(5):
        for j in range(5):
            if matrix[i][j] == ch: return i, j

def encrypt(text, matrix):
    text = text.upper().replace("J","I").replace(" ","")
    i = 0; cipher = ""
    while i < len(text):
        a = text[i]; b = 'X'
        if i + 1 < len(text): b = text[i + 1]
        if a == b: b = 'X'; i += 1
        else: i += 2
        r1,c1 = find_pos(matrix,a); r2,c2 = find_pos(matrix,b)
        if   r1==r2: cipher+=matrix[r1][(c1+1)%5]+matrix[r2][(c2+1)%5]
        elif c1==c2: cipher+=matrix[(r1+1)%5][c1]+matrix[(r2+1)%5][c2]
        else:         cipher+=matrix[r1][c2]+matrix[r2][c1]
    return cipher

def decrypt(text, matrix):
    plain = ""; i = 0
    while i < len(text):
        a=text[i]; b=text[i+1]; i+=2
        r1,c1=find_pos(matrix,a); r2,c2=find_pos(matrix,b)
        if   r1==r2: plain+=matrix[r1][(c1-1)%5]+matrix[r2][(c2-1)%5]
        elif c1==c2: plain+=matrix[(r1-1)%5][c1]+matrix[(r2-1)%5][c2]
        else:         plain+=matrix[r1][c2]+matrix[r2][c1]
    return plain

# RSA Encryption

def gcd(a, b):
    while b != 0:
        a, b = b, a % b
    return a

def mod_inverse(e, phi):
    for d in range(1, phi):
        if (d * e) % phi == 1:
            return d

def mod_exp(base, exp, mod):
    result = 1
    for i in range(exp):
        result = (result * base) % mod
    return result

p   = int(input("Enter p: "))
q   = int(input("Enter q: "))
n   = p * q
phi = (p - 1) * (q - 1)
e   = int(input("Enter e: "))

if gcd(e, phi) != 1:
    print("Invalid e"); exit()

d   = mod_inverse(e, phi)
msg = input("Enter message: ")

ascii_values    = [ord(char) for char in msg]
cipher          = [mod_exp(val, e, n) for val in ascii_values]
decrypted_ascii = [mod_exp(val, d, n) for val in cipher]
decrypted_msg   = ''.join(chr(val) for val in decrypted_ascii)

print("Public Key:",          (e, n))
print("Private Key:",         (d, n))
print("Encrypted Message:",   cipher)
print("Decrypted Message:",   decrypted_msg)

# MD5 Hash (Simplified)

def left_rotate(x, c):
    return ((x << c) | (x >> (32 - c))) & 0xFFFFFFFF

def simple_md5(message):
    A = 0x67452301
    B = 0xefcdab89
    C = 0x98badcfe
    D = 0x10325476

    msg = [ord(c) for c in message]

    for i in msg:
        F    = (B & C) | (~B & D)
        temp = (A + F + i) & 0xFFFFFFFF
        A, B, C, D = D, (B + left_rotate(temp, 5)) & 0xFFFFFFFF, B, C

    return hex(A)[2:] + hex(B)[2:] + hex(C)[2:] + hex(D)[2:]

def hex_to_binary_128(hex_str):
    binary = bin(int(hex_str, 16))[2:]
    return binary.zfill(128)

msg      = input("Enter message: ")
hash_hex = simple_md5(msg)
hash_bin = hex_to_binary_128(hash_hex)

print("MD5 Hash (Hex):",          hash_hex)
print("MD5 Hash (128-bit Binary):", hash_bin)
print("Length:", len(hash_bin), "bits")
if len(hash_bin) == 128:
    print("Verified: Output is exactly 128 bits")
else:
    print("Error: Output is not 128 bits")

# SHA-256 Hash (Simplified)

def right_rotate(x, n):
    return ((x >> n) | (x << (32 - n))) & 0xFFFFFFFF

def simple_sha256(message):
    h0 = 0x6a09e667;  h1 = 0xbb67ae85
    h2 = 0x3c6ef372;  h3 = 0xa54ff53a
    h4 = 0x510e527f;  h5 = 0x9b05688c
    h6 = 0x1f83d9ab;  h7 = 0x5be0cd19

    msg = [ord(c) for c in message]

    for i in msg:
        s1    = right_rotate(h4, 6) ^ right_rotate(h4, 11)
        ch    = (h4 & h5) ^ (~h4 & h6)
        temp1 = (h7 + s1 + ch + i) & 0xFFFFFFFF
        s0    = right_rotate(h0, 2) ^ right_rotate(h0, 13)
        maj   = (h0 & h1) ^ (h0 & h2) ^ (h1 & h2)
        temp2 = (s0 + maj) & 0xFFFFFFFF
        h7=h6; h6=h5; h5=h4
        h4 = (h3 + temp1) & 0xFFFFFFFF
        h3=h2; h2=h1; h1=h0
        h0 = (temp1 + temp2) & 0xFFFFFFFF

    return (hex(h0)[2:] + hex(h1)[2:] + hex(h2)[2:] + hex(h3)[2:] +
            hex(h4)[2:] + hex(h5)[2:] + hex(h6)[2:] + hex(h7)[2:])

def to_binary_256(hex_str):
    return bin(int(hex_str, 16))[2:].zfill(256)

msg      = input("Enter message: ")
hash_hex = simple_sha256(msg)
hash_bin = to_binary_256(hash_hex)

print("SHA-256 Hash (Hex):",    hash_hex)
print("SHA-256 Hash (Binary):", hash_bin)
print("Length:", len(hash_bin), "bits")
if len(hash_bin) == 256:
    print("Verified: Output is exactly 256 bits")
else:
    print("Error: Output is not 256 bits")

# Diffie-Hellman Key Exchange

def power(a, b, p):
    result = 1
    for i in range(b):
        result = (result * a) % p
    return result

def main():
    P = int(input("Enter value of P (prime): "))
    G = int(input("Enter value of G (primitive root): "))

    a = int(input("Enter private key of Alice: "))
    b = int(input("Enter private key of Bob: "))

    x = power(G, a, P)    # Alice's public key
    y = power(G, b, P)    # Bob's public key

    print("Public key of Alice:", x)
    print("Public key of Bob:",   y)

    K = power(y, a, P)    # Shared secret
    print("Shared Key:", K)

    msg          = input("Enter message: ")
    ascii_values = [ord(c) for c in msg]
    encrypted    = [val ^ K for val in ascii_values]
    print("Encrypted Message:", encrypted)

    decrypted_ascii = [val ^ K for val in encrypted]
    decrypted_msg   = ''.join(chr(val) for val in decrypted_ascii)
    print("Decrypted Message:", decrypted_msg)

main()
